206 inline RGB HSVtoRGB(float h, float s, float v) {
207 float r = 0, g = 0, b = 0;
208
209 int i = static_cast<int>(h * 6.0f);
210 float f = h * 6.0f - i;
211 float p = v * (1.0f - s);
212 float q = v * (1.0f - f * s);
213 float t = v * (1.0f - (1.0f - f) * s);
214
215 switch (i % 6) {
216 case 0: r = v; g = t; b = p; break;
217 case 1: r = q; g = v; b = p; break;
218 case 2: r = p; g = v; b = t; break;
219 case 3: r = p; g = q; b = v; break;
220 case 4: r = t; g = p; b = v; break;
221 case 5: r = v; g = p; b = q; break;
222 }
223
224 return RGB{
225 static_cast<int>(r * 255.0f),
226 static_cast<int>(g * 255.0f),
227 static_cast<int>(b * 255.0f)
228 };
229 }
