240std::vector<std::string> Split(const std::string& s, char seperator)
241{
242 std::vector<std::string> output;
243
244 std::string::size_type prev_pos = 0, pos = 0;
245
246 while ((pos = s.find(seperator, pos)) != std::string::npos)
247 {
248 std::string substring(s.substr(prev_pos, pos - prev_pos));
249
250 output.push_back(substring);
251
252 prev_pos = ++pos;
253 }
254
255 output.push_back(s.substr(prev_pos, pos - prev_pos)); // Last word
256
257 return output;
258}
